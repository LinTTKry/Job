# 10.二进制加法 剑指65

不用加减乘除做加法；

{% hint style="info" %}
不考虑进位的和运算为异或^;

进位为&

循环直到没有进位
{% endhint %}

```javascript
function add(num1, num2){
   let sum,carry;
   while(num2!=0){//直到没有进位
       sum = num1 ^ num2;
       carry = (num1 & num2)<<1 ;
       //向右移去掉一个进位；
       num1 = sum;
       num2 = carry
   }
   return num1
}
```

```javascript
/**
 * @param {string} a
 * @param {string} b
 * @return {string}
 */
const addBinary = (a, b) => {
   let len  = Math.max(a.length, b.length);
   a = a.padStart(len,0)
   b = b.padStart(len,0);
   let res = [len];
   let sum=0, car = 0;
   for(let i = len-1; i>=0; i--){
      sum = parseInt(a[i])+parseInt(b[i])+car;
      if(sum>=2){
         res[i] = sum-2;
         car = 1;
      }else{
         res[i] = sum;
         car = 0
      }
   }
   if(car) res.unshift(1);
   return res.join('')
}
```
