# 17.判断一个质数,计数质数

```javascript
function isPrime(n){
    for(let i = 2; i<= Math.sqrt(n);i++){
       if(n%i==0) return false
    }
    return true;
}
```

```javascript
/**
 * @param {number} n
 * @return {number}
 */
var countPrimes = function(n) {
    let res =0;
    let isP = new Array(n).fill(1);
    for(let i = 2; i<n;i++){
       if(isP[i]==1){
          res++;
          for(let j = i*i; j<n; j+=i) isP[j] = 0
       }
    }
}
```
