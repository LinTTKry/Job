# 05.重建二叉树

```javascript
function TreeNode(x) {
    this.val = x;
    this.left = null;
    this.right = null;
}
function reConstructBinaryTree(pre, vin)
{
    if(!pre.length || !vin.length) return 
    let root = new TreeNode(pre[0]);
    let vinrootIndex = vin.indexOf(pre[0]);
    root.left = reConstructBinaryTree(pre.slice(1,1+vinrootIndex), vin.slice(0,vinrootIndex))
    root.right = reConstructBinaryTree(pre.slice(1+vinrootIndex), vin.slice(vinrootIndex+1))
    return root
}
```
